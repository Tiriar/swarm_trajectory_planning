159  void solve(unsigned long int start){
160  long int node_idx;
161  CHeap<Nodes, Weight> heap(nodes.size(), nodes);
162  for (long int i = 0; i < nodes.size(); i++) {
163  nodes[i].weight = -1;
164  nodes[i].pred = -1;
165  }
166  nodes[start].weight = 0;
167  heap.add(start, 0);
168  while((node_idx = heap.getFirst()) != -1) {
169  SNode & node = nodes[node_idx];
170 
171  BOOST_FOREACH(SEdge e, node.edges) {
172  SNode & to = nodes[e.to];
173  if (heap.getIDX(e.to) == -1) {
174  if (to.weight == -1) { //not in close list
175  to.weight = node.weight + e.weight;
176  to.pred = node_idx;
177  heap.add(e.to, to.weight);
178  }
179  } else {
180  Weight c = node.weight + e.weight;
181  if (to.weight > c) {
182  to.weight = c;
183  to.pred = node_idx;
184  heap.update(e.to, to.weight);
185  }
186  }
187  } //end for loop
188  heap.getIDX(node_idx) = -1;
189  }
